Java对ES的操作示例

public Date getToDBTime(String orderCode, Date startTime, Date endTime) {
        SimpleDateFormat formatter = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSSXXX");
        BoolQueryBuilder boolQuery = new BoolQueryBuilder()
                .filter(QueryBuilders.rangeQuery("logTime").gte(startTime.getTime()).lte(endTime.getTime()))
                .filter(QueryBuilders.matchPhraseQuery("logMessage", "GetStuClassCodeByOrder GetStuClassCodeByOrder 根据订单:" + orderCode))
                .filter(QueryBuilders.queryStringQuery("log.file.path:RedisServiceOrdersToDB*"));
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder()
                .query(boolQuery);
        SearchRequest searchRequest = new SearchRequest("p0-gaobingfa1-*")
                .source(sourceBuilder);
        String time = "";
        try {
            SearchResponse searchResponse = client.search(searchRequest, RequestOptions.DEFAULT);
            SearchHits searchHits = searchResponse.getHits();
            for (SearchHit searchHit : searchHits.getHits()) {
                time = searchHit.getSourceAsMap().get("logTime").toString();
                return formatter.parse(time);
            }
        } catch (Exception e) {
            e.printStackTrace();
            return null;
        }
        return new Date();
    }
