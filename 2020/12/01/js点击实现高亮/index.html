js点击实现高亮

2020-12-01
知识点-前端
//在JS中添加
<script type="text/javascript">
var theObj;
function hilight(obj){
    if(theObj != null){
        theObj.style.background = "#FFF";
    }
    if(theObj = obj){
        obj.style.background = "#00FFFF";
    }
}
</script>

//在需要显示高亮的地方添加
<tr onclick="hilight(this)"> </tr>
